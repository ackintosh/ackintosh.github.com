        new Writer(list).start();

class Writer extends Thread {
    private final List<Integer> list;

    public Writer(List<Integer> list) {
        super("Writer");
        this.list = list;
    }

    public void run() {
        for (int i = 0; true; i++) {
            list.add(i);
            list.remove(0);
        }
    }
}

class Reader extends Thread {
    private final List<Integer> list;

    public Reader(List<Integer> list) {
        super("Reader");
        this.list = list;
    }

    public void run() {
        while (true) {
            for (int i : list) {
                System.out.println(i);
            }
        }
    }
}
